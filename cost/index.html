select *, average_cost_per_case/global_average_cost_per_case as cgw
from (
	select a.drg_code, drg_name, count(*) as case_count , avg(encounter_los) as alos, 
		stddev(encounter_los)*2 as 2sd, min(encounter_los) as min, 
		max(encounter_los) as max, avg(encounter_cost) as average_cost_per_case
	from cases a
	left join drg b on (a.drg_code = b.drg_code) 
	group by drg_code
) as a
cross join (
    select  avg(encounter_cost) as global_average_cost_per_case from cases
) as x
